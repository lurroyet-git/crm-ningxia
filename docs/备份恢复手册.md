## 二、备份脚本

### 2.1 数据库备份脚本

创建 `/opt/crm-platform/scripts/backup-database.sh`：

```bash
#!/bin/bash
# ==========================================
# 宁夏CRM数据库备份脚本
# 执行频率: 每日 02:00 (cron)
# ==========================================

set -e

# 配置
BACKUP_DIR="/backups/database"
DATE=$(date +%Y%m%d_%H%M%S)
DB_NAME="crm_production"
DB_USER="crm_user"
DB_CONTAINER="crm-postgres"
RETENTION_DAYS=7

# 创建备份目录
mkdir -p ${BACKUP_DIR}

# 备份文件名
BACKUP_FILE="${BACKUP_DIR}/db_backup_${DATE}.sql"
COMPRESSED_FILE="${BACKUP_FILE}.gz"

# 执行备份
echo "[$(date)] 开始数据库备份..."
sudo docker exec -i ${DB_CONTAINER} pg_dump \
  -U ${DB_USER} \
  -d ${DB_NAME} \
  --no-owner \
  --no-privileges \
  --clean \
  --if-exists \
  > ${BACKUP_FILE}

# 压缩备份文件
echo "[$(date)] 压缩备份文件..."
gzip -f ${BACKUP_FILE}

# 验证备份文件
if [ -f "${COMPRESSED_FILE}" ]; then
    FILE_SIZE=$(du -sh ${COMPRESSED_FILE} | cut -f1)
    echo "[$(date)] 备份成功: ${COMPRESSED_FILE} (${FILE_SIZE})"
else
    echo "[$(date)] 备份失败！"
    exit 1
fi

# 删除过期备份
echo "[$(date)] 清理${RETENTION_DAYS}天前的备份..."
find ${BACKUP_DIR} -name "db_backup_*.sql.gz" -mtime +${RETENTION_DAYS} -delete

# 备份统计
BACKUP_COUNT=$(find ${BACKUP_DIR} -name "db_backup_*.sql.gz" | wc -l)
echo "[$(date)] 当前备份数量: ${BACKUP_COUNT}"

# 上传异地备份（可选）
if command -v aws &> /dev/null; then
    echo "[$(date)] 上传至异地存储..."
    aws s3 cp ${COMPRESSED_FILE} s3://your-backup-bucket/crm/database/ 2>/dev/null || echo "异地备份上传失败（请检查AWS配置）"
fi

echo "[$(date)] 数据库备份完成"
```

```bash
chmod +x /opt/crm-platform/scripts/backup-database.sh
```

### 2.2 文件备份脚本

创建 `/opt/crm-platform/scripts/backup-files.sh`：

```bash
#!/bin/bash
# ==========================================
# 宁夏CRM文件备份脚本（MinIO）
# 执行频率: 每日 03:00 (cron)
# ==========================================

set -e

BACKUP_DIR="/backups/files"
DATE=$(date +%Y%m%d_%H%M%S)
MINIO_BUCKET="crm-files"
MINIO_CONTAINER="crm-minio"
RETENTION_DAYS=7

mkdir -p ${BACKUP_DIR}

BACKUP_PATH="${BACKUP_DIR}/minio_backup_${DATE}"

echo "[$(date)] 开始文件备份..."

# 使用mc命令备份MinIO存储桶
sudo docker exec -i ${MINIO_CONTAINER} sh -c "
  mc alias set local http://localhost:9000 \${MINIO_ROOT_USER} \${MINIO_ROOT_PASSWORD} &&
  mc mirror local/${MINIO_BUCKET} /backup/temp_mirror
"

# 打包备份
sudo docker cp ${MINIO_CONTAINER}:/backup/temp_mirror ${BACKUP_PATH}
tar -czf ${BACKUP_PATH}.tar.gz -C ${BACKUP_DIR} minio_backup_${DATE}
rm -rf ${BACKUP_PATH}

if [ -f "${BACKUP_PATH}.tar.gz" ]; then
    FILE_SIZE=$(du -sh ${BACKUP_PATH}.tar.gz | cut -f1)
    echo "[$(date)] 备份成功: ${BACKUP_PATH}.tar.gz (${FILE_SIZE})"
else
    echo "[$(date)] 备份失败！"
    exit 1
fi

# 清理过期备份
find ${BACKUP_DIR} -name "minio_backup_*.tar.gz" -mtime +${RETENTION_DAYS} -delete

BACKUP_COUNT=$(find ${BACKUP_DIR} -name "minio_backup_*.tar.gz" | wc -l)
echo "[$(date)] 当前备份数量: ${BACKUP_COUNT}"

echo "[$(date)] 文件备份完成"
```

```bash
chmod +x /opt/crm-platform/scripts/backup-files.sh
```

### 2.3 配置备份脚本

创建 `/opt/crm-platform/scripts/backup-config.sh`：

```bash
#!/bin/bash
# ==========================================
# 宁夏CRM配置备份脚本
# 执行频率: 每次配置变更后手动执行
# ==========================================

set -e

BACKUP_DIR="/backups/config"
DATE=$(date +%Y%m%d_%H%M%S)
BACKUP_FILE="${BACKUP_DIR}/config_backup_${DATE}.tar.gz"

mkdir -p ${BACKUP_DIR}

echo "[$(date)] 开始配置备份..."

# 打包所有配置文件
tar -czf ${BACKUP_FILE} \
  /opt/crm-platform/.env \
  /opt/crm-platform/docker-compose.prod.yml \
  /opt/crm-platform/prometheus.yml \
  /opt/crm-platform/alertmanager.yml \
  /opt/crm-platform/promtail-config.yml \
  /etc/nginx/sites-available/crm \
  /etc/letsencrypt/ 2>/dev/null || true

if [ -f "${BACKUP_FILE}" ]; then
    FILE_SIZE=$(du -sh ${BACKUP_FILE} | cut -f1)
    echo "[$(date)] 配置备份成功: ${BACKUP_FILE} (${FILE_SIZE})"
else
    echo "[$(date)] 配置备份失败！"
    exit 1
fi

# 保留最近30个配置备份
ls -t ${BACKUP_DIR}/config_backup_*.tar.gz | tail -n +31 | xargs -r rm -f

echo "[$(date)] 配置备份完成"
```

```bash
chmod +x /opt/crm-platform/scripts/backup-config.sh
```

### 2.4 一键全量备份脚本

创建 `/opt/crm-platform/scripts/backup-all.sh`：

```bash
#!/bin/bash
# ==========================================
# 宁夏CRM一键全量备份
# ==========================================

set -e

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
LOG_FILE="/var/log/crm-backup-$(date +%Y%m%d_%H%M%S).log"

exec > >(tee -a ${LOG_FILE}) 2>&1

echo "========================================"
echo "[$(date)] CRM全量备份开始"
echo "========================================"

${SCRIPT_DIR}/backup-database.sh
${SCRIPT_DIR}/backup-files.sh
${SCRIPT_DIR}/backup-config.sh

echo "========================================"
echo "[$(date)] CRM全量备份完成"
echo "日志文件: ${LOG_FILE}"
echo "========================================"
```

```bash
chmod +x /opt/crm-platform/scripts/backup-all.sh
```

## 四、恢复步骤

### 4.1 数据库恢复

#### 场景A：完整数据库恢复（最新备份）

```bash
# 1. 停止后端服务（防止写入）
sudo docker compose -f /opt/crm-platform/docker-compose.prod.yml stop backend

# 2. 备份当前数据库状态（以防万一）
sudo docker exec -i crm-postgres pg_dump -U crm_user crm_production > /backups/emergency_$(date +%Y%m%d_%H%M%S).sql

# 3. 删除并重建数据库（危险操作！请确认备份有效）
sudo docker exec -i crm-postgres psql -U crm_user -d postgres -c "DROP DATABASE IF EXISTS crm_production;"
sudo docker exec -i crm-postgres psql -U crm_user -d postgres -c "CREATE DATABASE crm_production;"

# 4. 恢复数据库
LATEST_BACKUP=$(ls -t /backups/database/db_backup_*.sql.gz | head -1)
echo "恢复文件: ${LATEST_BACKUP}"
gunzip -c ${LATEST_BACKUP} | sudo docker exec -i crm-postgres psql -U crm_user -d crm_production

# 5. 验证恢复
echo "用户表记录数:"
sudo docker exec -i crm-postgres psql -U crm_user -d crm_production -c "SELECT COUNT(*) FROM \"User\";"

echo "项目表记录数:"
sudo docker exec -i crm-postgres psql -U crm_user -d crm_production -c "SELECT COUNT(*) FROM \"Project\";"

echo "客户表记录数:"
sudo docker exec -i crm-postgres psql -U crm_user -d crm_production -c "SELECT COUNT(*) FROM \"Customer\";"

# 6. 重启后端服务
sudo docker compose -f /opt/crm-platform/docker-compose.prod.yml start backend

# 7. 验证服务
sleep 5
curl -f http://localhost:3001/api/health
echo "数据库恢复完成！"
```

#### 场景B：恢复到特定时间点（需启用WAL归档）

```bash
# 1. 停止后端服务
sudo docker compose -f /opt/crm-platform/docker-compose.prod.yml stop backend

# 2. 准备基础备份
# 3. 从归档恢复WAL到指定时间点
# 4. 启动数据库
# 5. 验证并启动后端

# 注意：WAL归档恢复需要PostgreSQL提前配置 archive_mode = on
# 详细配置请参考PostgreSQL官方文档
```

#### 场景C：单表恢复

```bash
# 1. 从备份中提取指定表数据
# 使用pg_restore的-t参数或手动提取INSERT语句

# 2. 先创建临时数据库恢复
sudo docker exec -i crm-postgres psql -U crm_user -d postgres -c "CREATE DATABASE crm_temp;"

gunzip -c /backups/database/db_backup_20260101_020000.sql.gz | sudo docker exec -i crm-postgres psql -U crm_user -d crm_temp

# 3. 导出需要的表数据
sudo docker exec -i crm-postgres pg_dump -U crm_user -d crm_temp -t '"Customer"' --data-only > /tmp/customer_restore.sql

# 4. 导入到生产数据库
sudo docker exec -i crm-postgres psql -U crm_user -d crm_production < /tmp/customer_restore.sql

# 5. 清理临时数据库
sudo docker exec -i crm-postgres psql -U crm_user -d postgres -c "DROP DATABASE crm_temp;"
```

### 4.2 文件恢复（MinIO）

```bash
# 1. 进入MinIO容器
sudo docker exec -it crm-minio sh

# 2. 使用mc恢复存储桶
mc alias set local http://localhost:9000 ${MINIO_ROOT_USER} ${MINIO_ROOT_PASSWORD}

# 从本地备份恢复
mc mirror /backups/files/minio_backup_20260101_030000 local/crm-files

# 或从异地备份恢复
mc mirror oss://your-backup-bucket/crm/files/latest local/crm-files

# 3. 验证文件列表
mc ls local/crm-files
```

### 4.3 配置恢复

```bash
# 1. 解压配置备份
tar -xzf /backups/config/config_backup_20260101_120000.tar.gz -C /tmp/config_restore

# 2. 恢复环境变量
sudo cp /tmp/config_restore/opt/crm-platform/.env /opt/crm-platform/.env

# 3. 恢复Nginx配置
sudo cp /tmp/config_restore/etc/nginx/sites-available/crm /etc/nginx/sites-available/crm
sudo nginx -t && sudo systemctl reload nginx

# 4. 恢复SSL证书（如需要）
sudo cp -r /tmp/config_restore/etc/letsencrypt/ /etc/

# 5. 重启Docker服务
sudo docker compose -f /opt/crm-platform/docker-compose.prod.yml restart
```

## 六、备份保留策略

### 6.1 保留策略表

| 备份类型 | 本地保留 | 异地保留 | 清理策略 | 说明 |

| 数据库全量 | 7天 | 30天 | 自动（cron） | 每日02:00生成 |

### 6.2 清理脚本

创建 `/opt/crm-platform/scripts/cleanup-backups.sh`：

```bash
#!/bin/bash
# ==========================================
# 备份清理脚本
# 执行频率: 每日 04:00
# ==========================================

set -e

# 数据库备份清理（保留7天）
echo "[$(date)] 清理数据库备份..."
find /backups/database -name "db_backup_*.sql.gz" -mtime +7 -delete

# 文件备份清理（保留7天）

echo "[$(date)] 清理配置备份..."
find /backups/config -name "config_backup_*.tar.gz" -mtime +30 -delete

# 日志清理（保留30天）
echo "[$(date)] 清理日志文件..."
find /var/log -name "crm-*.log" -mtime +30 -delete

echo "[$(date)] 备份清理完成"
```

```bash
find /backups/database -name "db_backup_*.sql.gz" | wc -l
find /backups/files -name "minio_backup_*.tar.gz" | wc -l

# 验证备份文件完整性（gzip测试）
gzip -t /backups/database/db_backup_20260101_020000.sql.gz && echo "备份文件有效" || echo "备份文件损坏！"

# 查看备份日志
tail -n 50 /var/log/crm-backup-db.log
```

### 7.2 自动化健康检查脚本

创建 `/opt/crm-platform/scripts/backup-health-check.sh`：

```bash
#!/bin/bash
# ==========================================
# 备份健康检查脚本
# 可集成到监控告警系统
# ==========================================

ERRORS=0

# 检查数据库备份是否存在（最近24小时内）
if [ ! "$(find /backups/database -name "db_backup_*.sql.gz" -mtime -1)" ]; then
    echo "ERROR: 最近24小时内无数据库备份！"
    ERRORS=$((ERRORS + 1))
else
    echo "OK: 数据库备份正常"
fi

# 检查文件备份是否存在
if [ ! "$(find /backups/files -name "minio_backup_*.tar.gz" -mtime -1)" ]; then
    echo "ERROR: 最近24小时内无文件备份！"
    ERRORS=$((ERRORS + 1))
else
    echo "OK: 文件备份正常"
fi

# 检查备份文件是否为空
EMPTY_BACKUPS=$(find /backups -name "*.gz" -size 0)
if [ -n "$EMPTY_BACKUPS" ]; then
    echo "ERROR: 发现空备份文件:"
    echo "$EMPTY_BACKUPS"
    ERRORS=$((ERRORS + 1))
fi

# 检查磁盘空间
DISK_USAGE=$(df /backups | awk 'NR==2 {print $5}' | sed 's/%//')
if [ "$DISK_USAGE" -gt 85 ]; then
    echo "WARNING: 备份磁盘使用率 ${DISK_USAGE}% > 85%"
    ERRORS=$((ERRORS + 1))
else
    echo "OK: 备份磁盘使用率 ${DISK_USAGE}%"
fi

if [ $ERRORS -gt 0 ]; then
    echo "检查失败，发现 $ERRORS 个问题"
    exit 1
else
    echo "所有检查通过"
    exit 0
fi
```

```bash
chmod +x /opt/crm-platform/scripts/backup-health-check.sh
```
